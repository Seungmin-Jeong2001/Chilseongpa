infra/ansible/
├── ansible.cfg
├── inventory.ini
├── playbook.yml          # ← GCP, AWS 역할을 배분하는 메인 대본
├── group_vars/           # ← 서버 그룹별로 다른 변수 저장
│   ├── all.yml           # 공통 변수 (프로젝트 명 등)
│   ├── gcp.yml           # GCP 전용 (DB 주소 등)
│   └── aws.yml           # AWS 전용 (AMI ID 등)
└── roles/
    ├── k3s_primary/      # GCP 전용 배역
    ├── k3s/              # AWS(Worker) 전용 배역
    ├── node-exporter/    # 공통 배역
    ├── cloudflared/      # 공통 배역
    └── monitoring/       # 모니터링 서버 전용 배역

playbook.yml
 # 1. 모든 서버 공통 작업
- hosts: all
  roles:
    - node-exporter
    - cloudflared

# 2. GCP 서버 (Primary K3s)
- hosts: gcp
  roles:
    - k3s_primary

# 3. AWS 서버 (Worker K3s + Monitoring)
- hosts: aws
  roles:
    - k3s
    - monitoring
